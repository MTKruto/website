### Result 

<div class="font-mono"><span>void</span></div>

### Syntax

```ts
// Required parameters only.
await client.promoteChatMember(chatId, userId);

// Required parameters + optional parameters.
// Any of the optional parameters can be omitted.
await client.promoteChatMember(chatId, userId, {
    isAnonymous,
    canManageChat,
    canDeleteMessages,
    canManageVideoChats,
    canRestrictMembers,
    canPromoteMembers,
    canChangeInfo,
    canInviteUsers,
    canPostMessages,
    canEditMessages,
    canPinMessages,
    canManageTopics,
    canPostStories,
    canEditStories,
    canDeleteStories,
    canManageDirectMessages,
    title,
});
```
