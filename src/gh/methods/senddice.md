```ts
// Required parameters only.

// Required parameters + optional parameters.
// Any of the optional parameters can be omitted.
await client.sendDice(chatId, {

    disableNotification,
    protectContent,
    replyTo,
    messageThreadId,
    sendAs,
    effectId,
    sendAt,
    businessConnectionId,
    replyMarkup,
});
```
